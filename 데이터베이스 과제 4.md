```SQL
CREATE TABLE Beer
(
    # 해당 테이블은 BeerID, BeerName, ManufactureName, Price 컬럼을 가진다.
    # BeerID는 PK이다.
    BeerName        VARCHAR(255) NOT NULL,
    BeerID          INT         NOT NULL AUTO_INCREMENT,
    ManufactureName VARCHAR(255) NOT NULL,
    Price           INT         NOT NULL,
    PRIMARY KEY (BeerID)
);

CREATE TABLE Drinker
(
    # 해당 테이블은 CitizenID, Name, Address, DOB, PhoneNumber, Gender 컬럼을 가진다.
    # CitizenID는 PK이다.
    CitizenID INT         NOT NULL AUTO_INCREMENT,
    Name      VARCHAR(20) NOT NULL,
    Address   VARCHAR(20) NOT NULL,
    DOB       DATE        NOT NULL,
    Gender    VARCHAR(20) NOT NULL,
    PRIMARY KEY (CitizenID)
);

CREATE TABLE Pub
(
    # 해당 테이블은 BussinessId, PubName, Address, PhoneNumber 컬럼을 가진다.
    # BussinessId는 PK이다.
    BussinessId INT         NOT NULL AUTO_INCREMENT,
    PubName     VARCHAR(20) NOT NULL,
    Address     VARCHAR(20) NOT NULL,
    PhoneNumber VARCHAR(20) NOT NULL,
    PRIMARY KEY (BussinessId)
);
```
